    agent any

    stages {
        stage('Checkoutttt') {
            steps {
                // Checkout source code
                checkout scm
            }
        }

        stage('Backend Build and Deploy') {
            steps {
                // Build the backend
                sh script 'mvn clean install'
            }
        }

        stage('Frontend Build and Deploy') {
            steps {
                // Build and deploy the frontend
                sh 'frontend-build-and-deploy-command'
            }
        }
    }
}
